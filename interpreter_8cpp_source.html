106void Interpreter::parseLine(const std::string &line, LineNumber &lineNum,
107 std::string &code) {
108 // Check if line starts with a number
109 size_t pos = 0;
110 while (pos < line.length() && std::isspace(line[pos]))
111 pos++;
112
113 if (pos < line.length() && std::isdigit(line[pos])) {
114 // Extract line number
115 size_t numEnd = pos;
116 while (numEnd < line.length() && std::isdigit(line[numEnd]))
117 numEnd++;
118
119 lineNum = std::stoi(line.substr(pos, numEnd - pos));
120
121 // Rest is code
122 while (numEnd < line.length() && std::isspace(line[numEnd]))
123 numEnd++;
124 code = line.substr(numEnd);
125 } else {
126 lineNum = -1;
127 code = line;

286void Interpreter::runFrom(LineNumber lineNum) {
287 running_ = true;
288 immediate_ = false;
289 resetOutputPosition();
290
291 // Prepare DATA cache before execution so READ works regardless of control
292 // flow. This scans all program lines and collects DATA statement values
293 // into a linear array (dataValues_) with an index mapping line numbers
294 // to positions (dataOffsets_). This allows RESTORE to reposition the
295 // data pointer to a specific line's data.
296 dataPointer_ = 0;
297 dataValues_.clear();
298 dataOffsets_.clear();
299 for (const auto &pair : program_) {
300 bool recorded = false;
301 for (const auto &stmt : pair.second.statements) {
302 size_t before = dataValues_.size();
303 stmt->collectData(dataValues_);
304 // Record the offset of the first DATA statement in this line
305 if (!recorded && dataValues_.size() > before) {
306 dataOffsets_.push_back({pair.first, before});
307 recorded = true;
308 }
309 }
310 }
311
312 // Set initial program counter position
313 if (lineNum < 0 && !program_.empty()) {
314 // Start from first line (RUN with no argument)
315 programCounter_ = program_.begin();
316 } else {
317 // Start from specified line (RUN linenum or GOTO)
318 programCounter_ = program_.find(lineNum);
319 if (programCounter_ == program_.end()) {
320 std::cout << "?UNDEF'D STATEMENT ERROR\n";
321 return;
322 }
323 }
324
325 try {
326 // Main execution loop: iterate through program lines
327 while (running_ && programCounter_ != program_.end()) {
328 currentLine_ = programCounter_->first;
329 jumped_ = false;
330
331 // TRACE output if enabled: show line number before execution
332 if (tracing_) {
333 std::cout << "[" << currentLine_ << "]";
334 }
335
336 try {
337 // Execute all statements on this line
338 for (auto &stmt : programCounter_->second.statements) {
339 stmt->execute(this);
340 // Stop executing statements if we've jumped or stopped
341 if (!running_ || jumped_)
342 break;
343 // Apply SPEED delay between statements if configured
344 applySpeedDelay();
345 }
346 } catch (const std::exception &e) {
347 // Error occurred - check if we have an error handler
348 if (errorHandlerLine_ >= 0) {
349 // ONERR handler is active - prepare error state
350 errorLine_ = currentLine_;
351 lastError_ = e.what();
352
353 // Store error information in memory locations for PEEK access
354 // These locations match Applesoft BASIC conventions:
355 // Location 218-219 (0xDA-0xDB): error line number (little-endian)
356 // Location 222 (0xDE): error code
357 pokeMemory(0x00DA, errorLine_ & 0xFF);
358 pokeMemory(0x00DB, (errorLine_ >> 8) & 0xFF);
359
360 // Only set generic error code if no specific error code was set
361 // by handleError() or other error-generating code
362 int currentErrorCode = peekMemory(0x00DE);
363 if (currentErrorCode == 0) {
364 pokeMemory(0x00DE, 16); // Generic error code
365 }
366
367 // Jump to error handler line
368 gotoLine(errorHandlerLine_);
369 continue; // Continue executing from error handler
370 } else {
371 // No error handler - print error and stop
372 std::cout << "?" << e.what() << " IN LINE " << currentLine_ << "\n";
373 running_ = false;
374 break;
375 }
376 }
377
378 // Advance to next line if we didn't jump
379 // (GOTO, GOSUB, or control flow statements set jumped_ flag)
380 if (!jumped_) {
381 ++programCounter_;
382 }
383 }
384 } catch (const std::exception &e) {
385 // Catch any unhandled exceptions from the main execution loop
386 // This is a safety net for errors that escape the inner try-catch
387 std::cout << "?" << e.what() << "\n";
388 running_ = false;
389 }
390
391 // Ensure execution state is clean after run completes
392 running_ = false;
393 paused_ = false;
394}

1872void Interpreter::callAddress(int address) {
1873 // Handle negative addresses (Apple II convention)
1874 if (address < 0) {
1875 address = 0x10000 + address;
1876 }
1877
1878 // Handle special Apple II CALL addresses
1879 switch (address) {
1880 case 0xF308: // -3288: Stack cleanup routine
1881 // No-op in our implementation
1882 break;
1883
1884 case 0xF3D2: // -3086: Clear hi-res page to black
1885 // Would clear graphics buffer in full implementation
1886 std::cout << "CALL $F3D2: CLEAR HI-RES TO BLACK (STUB)\n";
1887 break;
1888
1889 case 0xF3D6: // -3082: Clear hi-res to last HPLOT color
1890 std::cout << "CALL $F3D6: CLEAR HI-RES TO COLOR (STUB)\n";
1891 break;
1892
1893 case 0xF832: // -1998: BKGND (background color)
1894 std::cout << "CALL $F832: SET BACKGROUND (STUB)\n";
1895 break;
1896
1897 case 0xFC42: // -958: Clear from cursor to bottom-right
1898 std::cout << "CALL $FC42: CLEAR TO BOTTOM (STUB)\n";
1899 break;
1900
1901 case 0xFC58: // -936: HOME (clear screen, home cursor)
1902 std::cout << "\033[2J\033[H"; // ANSI clear screen
1903 outputRow_ = 0;
1904 pokeMemory(0x0025, outputRow_);
1905 break;
1906
1907 case 0xFC66: // -922: Line feed
1908 std::cout << "\n";
1909 outputRow_++;
1910 if (outputRow_ >= 24)
1911 outputRow_ = 23;
1912 pokeMemory(0x0025, outputRow_);
1913 break;
1914
1915 case 0xFC70: // -912: Scroll text window up
1916 std::cout << "CALL $FC70: SCROLL UP (STUB)\n";
1917 break;
1918
1919 case 0xFC9C: // -868: CLREOL (clear to end of line)
1920 std::cout << "\033[K"; // ANSI clear to end of line
1921 break;
1922
1923 case 0xFF69: // -151: Enter Monitor
1924 std::cout << "CALL $FF69: MONITOR (NOT IMPLEMENTED)\n";
1925 break;
1926
1927 case 0x0300: // Common user ML routine location (page 3)
1928 std::cout << "CALL $0300: USER ROUTINE (STUB)\n";
1929 break;
1930
1931 case 0x03EA: // Restore ProDOS connection
1932 std::cout << "CALL $03EA: RESTORE PRODOS (STUB)\n";
1933 break;
1934
1935 default:
1936 // Generic machine language call - no-op
1937 std::cout << "CALL " << formatHexAddress(address) << " (NOT IMPLEMENTED)\n";
1938 break;
1939 }
