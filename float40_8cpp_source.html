60std::string Float40::toString() const {
61 std::ostringstream oss;
62
63 // Match Applesoft output format
64 if (std::abs(value_) >= 1e9 || (std::abs(value_) < 0.01 && value_ != 0.0)) {
65 // Scientific notation
66 oss << std::scientific << std::setprecision(8) << value_;
67 } else {
68 // Fixed notation
69 oss << std::fixed << std::setprecision(9) << value_;
70 std::string result = oss.str();
71
72 // Remove trailing zeros
73 size_t dot = result.find('.');
74 if (dot != std::string::npos) {
75 size_t end = result.find_last_not_of('0');
76 if (end > dot) {
77 result = result.substr(0, end + 1);
78 }
79 if (result.back() == '.') {
80 result.pop_back();
81 }
82 }
83 return result;
84 }

89double Float40::roundToPrecision() const {
