52std::string Variables::normalizeName(const std::string &name) const {
53 // In Applesoft BASIC, only first 2 characters are significant
54 std::string normalized = name;
55 std::transform(normalized.begin(), normalized.end(), normalized.begin(),
56 ::toupper);
57
58 // Preserve two significant characters in the name part of user-defined functions (FNxy)
59 if (normalized.rfind("FN", 0) == 0 && normalized.length() > 2) {
60 return normalized.substr(0, std::min(normalized.length(), FN_FUNCTION_NAME_LENGTH));
61 }
62
63 // For non-string, non-integer variables, use first 2 chars
64 if (normalized.length() > VARIABLE_NAME_LENGTH && normalized.back() != '$' &&
65 normalized.back() != '%') {
66 return normalized.substr(0, VARIABLE_NAME_LENGTH);
67 }
68
69 return normalized;
70}

83Value coerceInteger(const Value &value) {
84 double n = value.getNumber();
85 // Applesoft integer variables are 16-bit signed; clamp to range.
86 double clamped = std::llround(n);
87 if (clamped > 32767.0)
88 clamped = 32767.0;
89 if (clamped < -32768.0)
90 clamped = -32768.0;
91 return Value(clamped);
92}

261void Variables::setArrayElement(const std::string &name,
262 const std::vector<int> &indices,
263 const Value &value) {
264 std::string normalized = normalizeName(name);
265
266 if (arrays_.find(normalized) == arrays_.end()) {
267 // Auto-dimension to 10 if not explicitly dimensioned (Applesoft behavior)
268 std::vector<int> defaultDims(indices.size(), 10);
269 dimArray(name, defaultDims);
270 }
271
272 ArrayInfo &arr = arrays_[normalized];
273
274 // Check bounds
275 for (size_t i = 0; i < indices.size() && i < arr.dimensions.size(); ++i) {
276 if (indices[i] < 0 || indices[i] > arr.dimensions[i]) {
277 throw std::runtime_error("BAD SUBSCRIPT ERROR");
278 }
279 }
280
281 if (!name.empty() && name.back() == '%') {
282 arr.data[indices] = coerceInteger(value);
283 } else {
284 arr.data[indices] = value;
285 }
286}
