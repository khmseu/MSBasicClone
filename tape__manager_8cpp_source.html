234void TapeManager::writeRecord(const std::vector<uint8_t> &data) {
235 if (!isOpen() || readMode_) {
236 throw std::runtime_error("TAPE NOT OPEN FOR WRITING");
237 }
238
239 // Write record size as 4-byte integer in native byte order
240 // Note: For cross-platform compatibility, tape files should be used on the
241 // same architecture or implement explicit little-endian serialization
242 uint32_t size = static_cast<uint32_t>(data.size());
243 file_.write(reinterpret_cast<const char *>(&size), sizeof(size));
244
245 // Write record data
246 file_.write(reinterpret_cast<const char *>(data.data()), data.size());
247
248 if (!file_) {
249 throw std::runtime_error("TAPE WRITE ERROR");
250 }
251
252 // Flush to ensure data is written to disk
253 file_.flush();
254
255 updatePosition();
256}

309std::vector<uint8_t> TapeManager::readRecord() {
310 if (!isOpen() || !readMode_) {
311 throw std::runtime_error("TAPE NOT OPEN FOR READING");
312 }
313
314 // Read record size
315 uint32_t size;
316 file_.read(reinterpret_cast<char *>(&size), sizeof(size));
317
318 if (!file_ || file_.eof()) {
319 throw std::runtime_error("END OF TAPE");
320 }
321
322 if (size == 0 || size > MAX_RECORD_SIZE) {
323 throw std::runtime_error("INVALID TAPE FORMAT");
324 }
325
326 // Read record data
327 std::vector<uint8_t> data(size);
328 file_.read(reinterpret_cast<char *>(data.data()), size);
329
330 if (!file_ || static_cast<size_t>(file_.gcount()) != size) {
331 throw std::runtime_error("TAPE READ ERROR");
332 }
333
334 updatePosition();
335 return data;
336}
