44std::vector<Token> Tokenizer::tokenize(const std::string &line) {
45 input_ = line;
46 pos_ = 0;
47 column_ = 1;
48
49 std::vector<Token> tokens;
50
51 while (!isAtEnd()) {
52 skipWhitespace();
53 if (isAtEnd())
54 break;
55
56 Token token = nextToken();
57 if (token.type != TokenType::NEWLINE || tokens.empty() ||
58 tokens.back().type != TokenType::NEWLINE) {
59 tokens.push_back(token);
60 }
61 }
62
63 return tokens;
64}

393Token Tokenizer::nextToken() {
394 char ch = peek();
395
396 // Numbers
397 if (std::isdigit(ch) || (ch == '.' && pos_ + 1 < input_.length() &&
398 std::isdigit(input_[pos_ + 1]))) {
399 return readNumber();
400 }
401
402 // Strings
403 if (ch == '"') {
404 return readString();
405 }
406
407 // Identifiers and keywords
408 if (std::isalpha(ch)) {
409 return readIdentifier();
410 }
411
412 // Operators and delimiters
413 return readOperator();
414}

431Token Tokenizer::readNumber() {
432 Token token;
433 token.type = TokenType::NUMBER;
434 token.line = line_;
435 token.column = column_;
436
437 std::string numStr;
438 bool hasDecimal = false;
439 bool hasExponent = false;
440
441 while (!isAtEnd()) {
442 char ch = peek();
443
444 if (std::isdigit(ch)) {
445 numStr += advance();
446 } else if (ch == '.' && !hasDecimal && !hasExponent) {
447 hasDecimal = true;
448 numStr += advance();
449 } else if ((ch == 'E' || ch == 'e') && !hasExponent) {
450 hasExponent = true;
451 numStr += advance();
452 if (!isAtEnd() && (peek() == '+' || peek() == '-')) {
453 numStr += advance();
454 }
455 } else {
456 break;
457 }
458 }
459
460 token.text = numStr;
461 token.value = Value(std::stod(numStr));
462 return token;
463}

526Token Tokenizer::readIdentifier() {
527 Token token;
528 token.line = line_;
529 token.column = column_;
530
531 std::string ident;
532 while (!isAtEnd() &&
533 (std::isalnum(peek()) || peek() == '$' || peek() == '%')) {
534 ident += advance();
535 }
536
537 // Check for function names with $
538 if (!ident.empty() && ident.back() == '$') {
539 std::string funcName = ident;
540 std::transform(funcName.begin(), funcName.end(), funcName.begin(),
541 ::toupper);
542
543 if (funcName == "CHR$" || funcName == "LEFT$" || funcName == "RIGHT$" ||
544 funcName == "MID$" || funcName == "STR$") {
545 token.type = getKeywordType(funcName);
546 token.text = funcName;
547 return token;
548 }
549 }
550
551 token.text = ident;
552
553 if (isKeyword(ident)) {
554 token.type = getKeywordType(ident);
555 } else {
556 token.type = TokenType::IDENTIFIER;
557 }
558
559 return token;
560}

582Token Tokenizer::readOperator() {
583 Token token;
584 token.line = line_;
585 token.column = column_;
586
587 char ch = peek();
588
589 switch (ch) {
590 case '+':
591 token.type = TokenType::PLUS;
592 token.text = "+";
593 advance();
594 break;
595 case '-':
596 token.type = TokenType::MINUS;
597 token.text = "-";
598 advance();
599 break;
600 case '*':
601 token.type = TokenType::MULTIPLY;
602 token.text = "*";
603 advance();
604 break;
605 case '/':
606 token.type = TokenType::DIVIDE;
607 token.text = "/";
608 advance();
609 break;
610 case '^':
611 token.type = TokenType::POWER;
612 token.text = "^";
613 advance();
614 break;
615 case '(':
616 token.type = TokenType::LPAREN;
617 token.text = "(";
618 advance();
619 break;
620 case ')':
621 token.type = TokenType::RPAREN;
622 token.text = ")";
623 advance();
624 break;
625 case ',':
626 token.type = TokenType::COMMA;
627 token.text = ",";
628 advance();
629 break;
630 case ';':
631 token.type = TokenType::SEMICOLON;
632 token.text = ";";
633 advance();
634 break;
635 case ':':
636 token.type = TokenType::COLON;
637 token.text = ":";
638 advance();
639 break;
640 case '#':
641 token.type = TokenType::HASH;
642 token.text = "#";
643 advance();
644 break;
645 case '$':
646 token.type = TokenType::DOLLAR;
647 token.text = "$";
648 advance();
649 break;
650 case '%':
651 token.type = TokenType::PERCENT;
652 token.text = "%";
653 advance();
654 break;
655 case '\n':
656 token.type = TokenType::NEWLINE;
657 token.text = "\\n";
658 advance();
659 break;
660
661 case '=':
662 token.text = "=";
663 advance();
664 token.type = TokenType::EQUAL;
665 break;
666
667 case '<':
668 token.text = "<";
669 advance();
670 if (!isAtEnd()) {
671 if (peek() == '=') {
672 token.text += "=";
673 token.type = TokenType::LESS_EQUAL;
674 advance();
675 } else if (peek() == '>') {
676 token.text += ">";
677 token.type = TokenType::NOT_EQUAL;
678 advance();
679 } else {
680 token.type = TokenType::LESS;
681 }
682 } else {
683 token.type = TokenType::LESS;
684 }
685 break;
686
687 case '>':
688 token.text = ">";
689 advance();
690 if (!isAtEnd() && peek() == '=') {
691 token.text += "=";
692 token.type = TokenType::GREATER_EQUAL;
693 advance();
694 } else {
695 token.type = TokenType::GREATER;
696 }
697 break;
698
699 default:
700 // Unknown character, skip it
701 token.type = TokenType::NEWLINE;
702 token.text = std::string(1, ch);
703 advance();
704 break;
705 }
706
707 return token;
708}
