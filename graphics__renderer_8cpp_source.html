262void GraphicsRenderer::drawText(const std::string &text, int x, int y,
263 int color) {
264#ifdef HAVE_RAYLIB
265 if (initialized_) {
266 Color c = {static_cast<unsigned char>((color >> 16) & 0xFF),
267 static_cast<unsigned char>((color >> 8) & 0xFF),
268 static_cast<unsigned char>(color & 0xFF), 255};
269
270 if (fontLoaded_) {
271 // Use loaded Apple II font
272 // Apple II character cell is 7×8 pixels
273 float fontSize = 8.0f * config_.scaleFactor;
274 float spacing = 1.0f * config_.scaleFactor;
275
276 // Apply horizontal scaling for 80-column mode (0.5x)
277 float horizontalScale = 1.0f;
278 if (config_.textMode == TextMode::Text80) {
279 horizontalScale = 0.5f;
280 }
281
282 Vector2 position = {
283 static_cast<float>(x * config_.scaleFactor * horizontalScale),
284 static_cast<float>(y * config_.scaleFactor)};
285
286 // If we're in 80-column mode, we need to scale the text horizontally
287 if (horizontalScale != 1.0f) {
288 // Pre-allocate character buffer for efficiency
289 char charBuffer[2] = {'\0', '\0'};
290 // Draw each character with adjusted spacing for horizontal compression
291 float currentX = position.x;
292 for (const char &ch : text) {
293 charBuffer[0] = ch;
294 Vector2 glyphPos{currentX, position.y};
295 DrawTextEx(font_, charBuffer, glyphPos, fontSize, 0, c);
296 // In 80-column mode, characters are compressed horizontally
297 currentX += (7.0f * config_.scaleFactor * horizontalScale);
298 }
299 } else {
300 // Normal 40-column mode rendering
301 DrawTextEx(font_, text.c_str(), position, fontSize, spacing, c);
302 }
303 } else {
304 // Fallback to built-in font
305 DrawText(text.c_str(), x * config_.scaleFactor, y * config_.scaleFactor,
306 10 * config_.scaleFactor, c);
307 }
308 }
309#endif
310}

374void GraphicsRenderer::loadApple2Font() {
375#ifdef HAVE_RAYLIB
376 // Search for the Ultimate Apple II Font in common locations
377 const std::vector<const char *> fontPaths = {
378 "assets/fonts/PrintChar21.ttf", "../assets/fonts/PrintChar21.ttf",
379 "/usr/share/fonts/apple2/PrintChar21.ttf"};
380
381 bool fontFound = false;
382 const char *foundPath = nullptr;
383
384 for (const char *path : fontPaths) {
385 // Check if file exists - using access() for safer checking
386#ifdef _WIN32
387 if (_access(path, 0) == 0) {
388#else
389 if (access(path, F_OK) == 0) {
390#endif
391 foundPath = path;
392 fontFound = true;
393 break;
394 }
395 }
396
397 if (fontFound) {
398 try {
399 // Load font at 8 pixels (Apple II character cell is 7×8)
400 // We use 8 here as the fontSize to get proper glyph rendering
401 font_ = LoadFontEx(foundPath, 8, nullptr, 0);
402
403 // Check if font loaded successfully using IsFontReady()
404 if (IsFontValid(font_)) {
405 // Apply bilinear filtering for better quality when scaling
406 SetTextureFilter(font_.texture, TEXTURE_FILTER_BILINEAR);
407
408 fontLoaded_ = true;
409 std::cout << "Successfully loaded Apple II font from: " << foundPath
410 << "\n";
411 std::cout << "Font base size: " << font_.baseSize << " pixels\n";
412 } else {
413 std::cerr << "Warning: Font file found but failed to load from: "
414 << foundPath << "\n";
415 // Unload the font to ensure proper cleanup of any partially loaded
416 // resources
417 UnloadFont(font_);
418 fontLoaded_ = false;
419 }
420 } catch (...) {
421 std::cerr << "Warning: Exception while loading font from: " << foundPath
422 << "\n";
423 // Attempt to unload font in case of partial initialization
424 try {
425 UnloadFont(font_);
426 } catch (...) {
427 // Ignore cleanup errors
428 }
429 fontLoaded_ = false;
430 }
431 }
432
433 if (!fontLoaded_) {
434 std::cout << "Ultimate Apple II Font not found or failed to load - using "
435 "Raylib default font\n";
436 std::cout << "Download from: "
437 "https://www.kreativekorp.com/software/fonts/apple2/\n";
438 std::cout << "Place in: assets/fonts/PrintChar21.ttf\n";
439 }
440#endif
441}
