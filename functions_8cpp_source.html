152void pokeMemory(int addr, int val) {
153 auto &mem = memoryMap();
154
155 // Handle negative addresses (Apple II convention)
156 // Applesoft BASIC allows negative addresses as shorthand for high memory
157 // Example: -16384 (decimal) = 49152 (0xC000) - the keyboard input register
158 if (addr < 0) {
159 addr = 0x10000 + addr; // Convert negative to 16-bit unsigned
160 }
161
162 // Special addresses that bypass range checks and have special behavior
163
164 // Memory pointers and error handling (0-255 range)
165 // These zero-page locations are used by the interpreter for system state
166 if (addr == 0x0025 || addr == 0x0069 || addr == 0x006A ||
167 addr == 0x0073 || addr == 0x0074 || addr == 0x00D8 ||
168 addr == 0x00DA || addr == 0x00DB || addr == 0x00DE) {
169 mem[addr] = val & 0xFF;
170 return;
171 }
172
173 // Text window control (32-37)
174 // These locations control the text output window position and dimensions
175 if (addr >= 0x0020 && addr <= 0x0025) {
176 mem[addr] = val & 0xFF;
177 return;
178 }
179
180 // Shape table pointers (232-233)
181 // Low and high bytes of shape table memory address for DRAW/XDRAW
182 if (addr == 0x00E8 || addr == 0x00E9) {
183 mem[addr] = val & 0xFF;
184 return;
185 }
186
187 // Hi-res page pointers (103-104)
188 // Stores which hi-res page is currently active (page 1 or page 2)
189 if (addr == 0x0067 || addr == 0x0068) {
190 mem[addr] = val & 0xFF;
191 return;
192 }
193
194 // Graphics memory base addresses
195 // These are the base addresses for hi-res graphics pages
196 if (addr == 0x4000 || addr == 0x6000) {
197 mem[addr] = val & 0xFF;
198 return;
199 }
200
201 // Keyboard strobe (49168 / -16368)
202 // Writing any value clears the keyboard strobe (acknowledges key press)
203 // In our implementation this is a no-op since we don't buffer keyboard input
204 if (addr == 0xC010 || addr == -16368) {
205 // Clear keyboard strobe - no-op in our implementation
206 return;
207 }
208
209 // Display control switches (49232-49239)
210 // These soft switches control text/graphics mode and page selection
211 // We store them but interpretation happens elsewhere
212 if (addr >= 0xC050 && addr <= 0xC057) {
213 mem[addr] = val & 0xFF;
214 return;
215 }
216
217 // Annunciator outputs (49240-49247)
218 // General-purpose output bits, typically used for hardware control
219 if (addr >= 0xC058 && addr <= 0xC05F) {
220 mem[addr] = val & 0xFF;
221 return;
222 }
223
224 // Standard memory range check
225 // All other addresses must fall within LOMEM to HIMEM bounds
226 if (addr < gLomem || addr > gHimem) {
227 throw std::runtime_error("MEMORY RANGE ERROR: " + formatHexAddress(addr, false));
228 }
229 mem[addr] = val & 0xFF;
230}

265int peekMemory(int addr) {
266 auto &mem = memoryMap();
267
268 // Handle negative addresses (Apple II convention)
269 // Same conversion as POKE: -16384 becomes 49152 (0xC000)
270 if (addr < 0) {
271 addr = 0x10000 + addr; // Convert negative to 16-bit unsigned
272 }
273
274 // Special addresses that bypass range checks
275
276 // Keyboard input (49152 / -16384)
277 // On real Apple II, returns last key with bit 7 set if key available
278 // We return stored value or 0 (stub implementation)
279 if (addr == 0xC000 || addr == -16384) {
280 // Return last key pressed (stub - would need actual keyboard state)
281 return mem.count(0xC000) ? mem[0xC000] : 0;
282 }
283
284 // Button inputs (49249-49251 / -16287 to -16285)
285 // Returns button state: bit 7 set if pressed, clear if not
286 // Stub implementation always returns 0 (not pressed)
287 if (addr >= 0xC061 && addr <= 0xC063) {
288 // Return button state (stub - always unpressed)
289 return 0;
290 }
291
292 // Memory pointers and error handling
293 // Same zero-page locations as POKE, readable for inspection
294 if (addr == 0x0025 || addr == 0x0069 || addr == 0x006A ||
295 addr == 0x0073 || addr == 0x0074 || addr == 0x00D8 ||
296 addr == 0x00DA || addr == 0x00DB || addr == 0x00DE) {
297 auto it = mem.find(addr);
298 if (it == mem.end()) {
299 return 0;
300 }
301 return it->second & 0xFF;
302 }
303
304 // Text window control (32-37)
305 if (addr >= 0x0020 && addr <= 0x0025) {
306 auto it = mem.find(addr);
307 return (it != mem.end()) ? (it->second & 0xFF) : 0;
308 }
309
310 // Text window control (32-37)
311 // Control the text output window boundaries
312 if (addr >= 0x0020 && addr <= 0x0025) {
313 auto it = mem.find(addr);
314 return (it != mem.end()) ? (it->second & 0xFF) : 0;
315 }
316
317 // Shape table pointers (232-233)
318 // Low/high bytes of shape table address used by DRAW/XDRAW commands
319 if (addr == 0x00E8 || addr == 0x00E9) {
320 auto it = mem.find(addr);
321 return (it != mem.end()) ? (it->second & 0xFF) : 0;
322 }
323
324 // Hi-res page pointers (103-104)
325 // Indicates which hi-res graphics page is active
326 if (addr == 0x0067 || addr == 0x0068) {
327 auto it = mem.find(addr);
328 return (it != mem.end()) ? (it->second & 0xFF) : 0;
329 }
330
331 // Graphics memory base addresses
332 // Base addresses for hi-res page 1 (0x4000) and page 2 (0x6000)
333 if (addr == 0x4000 || addr == 0x6000) {
334 auto it = mem.find(addr);
335 return (it != mem.end()) ? (it->second & 0xFF) : 0;
336 }
337
338 // Display control switches (49232-49239)
339 // Soft switches that control display modes on Apple II
340 if (addr >= 0xC050 && addr <= 0xC057) {
341 auto it = mem.find(addr);
342 return (it != mem.end()) ? (it->second & 0xFF) : 0;
343 }
344
345 // Annunciator outputs (49240-49247)
346 // General-purpose output bits for peripheral control
347 if (addr >= 0xC058 && addr <= 0xC05F) {
348 auto it = mem.find(addr);
349 return (it != mem.end()) ? (it->second & 0xFF) : 0;
350 }
351
352 // Standard memory range check
353 // For addresses in normal RAM range, validate against LOMEM/HIMEM
354 if (addr < gLomem || addr > gHimem) {
355 throw std::runtime_error("MEMORY RANGE ERROR: " + formatHexAddress(addr, false));
356 }
357 // Return stored value or 0 if never written (sparse storage)
358 auto it = mem.find(addr);
359 if (it == mem.end()) {
360 return 0;
361 }
362 return it->second & 0xFF;
363}

881Value funcPos(const Value &) {
882 int col = -1;
883
884#ifdef _WIN32
885 HANDLE h = GetStdHandle(STD_OUTPUT_HANDLE);
886 if (h != nullptr && h != INVALID_HANDLE_VALUE) {
887 CONSOLE_SCREEN_BUFFER_INFO info;
888 if (GetConsoleScreenBufferInfo(h, &info)) {
889 col = static_cast<int>(info.dwCursorPosition.X);
890 }
891 }
892#else
893 if (isatty(STDIN_FILENO) && isatty(STDOUT_FILENO)) {
894 termios oldt;
895 if (tcgetattr(STDIN_FILENO, &oldt) == 0) {
896 termios raw = oldt;
897 raw.c_lflag &= static_cast<unsigned>(~(ICANON | ECHO));
898 raw.c_cc[VMIN] = 0;
899 raw.c_cc[VTIME] = 1; // Tenths of a second
900
901 if (tcsetattr(STDIN_FILENO, TCSANOW, &raw) == 0) {
902 const char query[] = "\x1b[6n";
903 (void)!write(STDOUT_FILENO, query, sizeof(query) - 1);
904
905 char buf[32];
906 ssize_t n = read(STDIN_FILENO, buf, sizeof(buf) - 1);
907 if (n > 0) {
908 buf[n] = '\0';
909 int row = 0;
910 int parsedCol = 0;
911 if (sscanf(buf, "\x1b[%d;%dR", &row, &parsedCol) == 2) {
912 col = parsedCol - 1; // Escape reports 1-based column
913 }
914 }
915
916 tcsetattr(STDIN_FILENO, TCSANOW, &oldt);
917 }
918 }
919 }
920#endif
921
922 if (col < 0) {
923 return Value(0.0);
924 }
925 return Value(static_cast<double>(col));
926}
