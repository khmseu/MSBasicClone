465class BinaryExpr : public Expression {
466public:
467 BinaryExpr(std::shared_ptr<Expression> left, TokenType op,
468 std::shared_ptr<Expression> right)
469 : left_(left), op_(op), right_(right) {}
470
471 Value evaluate(Interpreter *interp) override {
472 Value lval = left_->evaluate(interp);
473 Value rval = right_->evaluate(interp);
474
475 switch (op_) {
476 case TokenType::PLUS:
477 return lval + rval;
478 case TokenType::MINUS:
479 return lval - rval;
480 case TokenType::MULTIPLY:
481 return lval * rval;
482 case TokenType::DIVIDE:
483 return lval / rval;
484 case TokenType::POWER: {
485 Float40 a(lval.getNumber());
486 Float40 b(rval.getNumber());
487 return Value(a.power(b).toDouble());
488 }
489 case TokenType::MOD: {
490 Float40 a(lval.getNumber());
491 Float40 b(rval.getNumber());
492 return Value(a.mod(b).toDouble());
493 }
494 case TokenType::EQUAL:
495 return Value(lval == rval ? 1.0 : 0.0);
496 case TokenType::NOT_EQUAL:
497 return Value(lval != rval ? 1.0 : 0.0);
498 case TokenType::LESS:
499 return Value(lval < rval ? 1.0 : 0.0);
500 case TokenType::GREATER:
501 return Value(lval > rval ? 1.0 : 0.0);
502 case TokenType::LESS_EQUAL:
503 return Value(lval <= rval ? 1.0 : 0.0);
504 case TokenType::GREATER_EQUAL:
505 return Value(lval >= rval ? 1.0 : 0.0);
506 case TokenType::AND:
507 return Value((lval.getNumber() != 0 && rval.getNumber() != 0) ? 1.0
508 : 0.0);
509 case TokenType::OR:
510 return Value((lval.getNumber() != 0 || rval.getNumber() != 0) ? 1.0
511 : 0.0);
512 default:
513 return Value(0.0);
514 }

1129class WaitStmt : public Statement {
1130public:
1131 WaitStmt(std::shared_ptr<Expression> addr, std::shared_ptr<Expression> mask,
1132 std::shared_ptr<Expression> timeout = nullptr)
1133 : addr_(std::move(addr)), mask_(std::move(mask)),
1134 timeout_(std::move(timeout)) {}

1136 int a = static_cast<int>(addr_->evaluate(interp).getNumber());
1137 int m = static_cast<int>(mask_->evaluate(interp).getNumber());
1138 // Optional timeout in milliseconds; <=0 means no timeout
1139 int timeoutMs = 0;
1140 if (timeout_) {
1141 timeoutMs = static_cast<int>(timeout_->evaluate(interp).getNumber());
1142 if (timeoutMs < 0)
1143 timeoutMs = 0;
1144 }
1145
1146 auto start = std::chrono::steady_clock::now();
1147 while (true) {
1148 int val = peekMemory(a);
1149 if ((val & m) != 0)
1150 break;
1151 if (timeoutMs > 0) {
1152 auto elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(
1153 std::chrono::steady_clock::now() - start);
1154 if (elapsed.count() >= timeoutMs) {
1155 // Timeout expires silently (Applesoft allowed external
1156 // events/timeouts)
1157 break;
1158 }
1159 }
1160 // Yield to avoid busy-loop; in real Applesoft would poll I/O
1161 std::this_thread::sleep_for(std::chrono::milliseconds(1));
1162 }
1163 }

1612std::vector<std::shared_ptr<Statement>>
1613Parser::parse(const std::vector<Token> &tokens) {
1614 std::vector<std::shared_ptr<Statement>> statements;
1615 size_t pos = 0;
1616
1617 while (pos < tokens.size()) {
1618 // Skip newlines
1619 if (tokens[pos].type == TokenType::NEWLINE) {
1620 pos++;
1621 continue;
1622 }
1623
1624 // Check for statement separator (colon)
1625 // Multiple statements on one line: 10 PRINT "A": PRINT "B"
1626 if (tokens[pos].type == TokenType::COLON) {
1627 pos++;
1628 continue;
1629 }
1630
1631 // Parse next statement and add to result
1632 auto stmt = parseStatement(tokens, pos);
1633 if (stmt) {
1634 statements.push_back(stmt);
1635 }
1636 }
1637
1638 return statements;
1639}

1669std::shared_ptr<Statement>
1670Parser::parseStatement(const std::vector<Token> &tokens, size_t &pos) {
1671 if (pos >= tokens.size())
1672 return nullptr;
1673
1674 Token &token = const_cast<Token &>(tokens[pos]);
1675
1676 // Dispatch to appropriate statement parser based on token type
1677 switch (token.type) {
1678 case TokenType::PRINT:
1679 return parsePrint(tokens, pos);
1680 case TokenType::LET:
1681 return parseLetOrAssignment(tokens, pos);
1682 case TokenType::END:
1683 pos++;
1684 return std::make_shared<EndStmt>();
1685 case TokenType::IF:
1686 return parseIf(tokens, pos);
1687 case TokenType::GOTO:
1688 return parseGoto(tokens, pos);
1689 case TokenType::GOSUB:
1690 return parseGosub(tokens, pos);
1691 case TokenType::RETURN:
1692 pos++;
1693 return std::make_shared<ReturnStmt>();
1694 case TokenType::STOP:
1695 pos++;
1696 return std::make_shared<StopStmt>();
1697 case TokenType::HTAB: {
1698 pos++; // Skip HTAB
1699 auto col = parseExpression(tokens, pos);
1700 return std::make_shared<HtabStmt>(col);
1701 }
1702 case TokenType::VTAB: {
1703 pos++; // Skip VTAB
1704 auto row = parseExpression(tokens, pos);
1705 return std::make_shared<VtabStmt>(row);
1706 }
1707 case TokenType::INVERSE:
1708 pos++;
1709 return std::make_shared<InverseStmt>();
1710 case TokenType::NORMAL:
1711 pos++;
1712 return std::make_shared<NormalStmt>();
1713 case TokenType::FLASH:
1714 pos++;
1715 return std::make_shared<FlashStmt>();
1716 case TokenType::FOR:
1717 return parseFor(tokens, pos);
1718 case TokenType::NEXT:
1719 return parseNext(tokens, pos);
1720 case TokenType::INPUT:
1721 return parseInput(tokens, pos);
1722 case TokenType::DIM:
1723 return parseDim(tokens, pos);
1724 case TokenType::DATA:

2542// ============================================================================
2543// Expression Parsing with Operator Precedence
2544// ============================================================================
2545// The following functions implement recursive descent parsing for expressions
2546// with proper operator precedence. Each level of precedence has its own
2547// parse function that calls the next higher precedence level.
2548//
2549// Precedence hierarchy (lowest to highest):
2550// 1. OR (logical or)
2551// 2. AND (logical and)
2552// 3. NOT (logical not)
2553// 4. Relational (=, <>, <, >, <=, >=)
2554// 5. Additive (+, -)
2555// 6. Multiplicative (*, /, MOD)
2556// 7. Unary (+, -)
2557// 8. Power (^)
2558// 9. Primary (literals, variables, functions, parentheses)
2559//
2560// This structure ensures that operations bind correctly:
2561// 2 + 3 * 4 = 2 + (3 * 4) = 14 (not (2 + 3) * 4 = 20)
2562// X = 5 OR Y = 3 evaluates comparisons before OR
2563// ============================================================================
