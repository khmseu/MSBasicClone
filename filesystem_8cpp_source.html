52std::vector<FileInfo> listFiles(const std::string &path) {
53 std::vector<FileInfo> files;
54
55#ifdef PLATFORM_WINDOWS
56 WIN32_FIND_DATAA findData;
57 HANDLE hFind = FindFirstFileA((path + "\\*").c_str(), &findData);
58
59 if (hFind != INVALID_HANDLE_VALUE) {
60 do {
61 if (strcmp(findData.cFileName, ".") != 0 &&
62 strcmp(findData.cFileName, "..") != 0) {
63 FileInfo info;
64 info.name = findData.cFileName;
65 info.size = findData.nFileSizeLow;
66 info.isDirectory =
67 (findData.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY) != 0;
68 files.push_back(info);
69 }
70 } while (FindNextFileA(hFind, &findData));
71 FindClose(hFind);
72 }
73#else
74 DIR *dir = opendir(path.c_str());
75 if (dir) {
76 struct dirent *entry;
77 while ((entry = readdir(dir)) != nullptr) {
78 if (strcmp(entry->d_name, ".") != 0 && strcmp(entry->d_name, "..") != 0) {
79 FileInfo info;
80 info.name = entry->d_name;
81 info.isDirectory = entry->d_type == DT_DIR;
82
83 std::string fullPath = path + "/" + entry->d_name;
84 struct stat st;
85 if (stat(fullPath.c_str(), &st) == 0) {
86 info.size = st.st_size;
87 } else {
88 info.size = 0;
89 }
90
91 files.push_back(info);
92 }
93 }
94 closedir(dir);
95 }
96#endif
97
98 return files;
99}

258int FileManager::openFile(const std::string &filename, FileAccessMode mode) {
259 // Check if file is already open
260 auto it = filenameToHandle_.find(filename);
261 if (it != filenameToHandle_.end()) {
262 return it->second; // Return existing handle
263 }
264
265 int handle = nextHandle_++;
266 FileHandle &fh = handles_[handle];
267 fh.filename = filename;
268 fh.mode = mode;
269 fh.position = 0;
270 fh.stream = std::make_unique<std::fstream>();
271
272 std::ios_base::openmode openMode = std::ios_base::binary;
273 switch (mode) {
274 case FileAccessMode::READ:
275 openMode |= std::ios_base::in;
276 break;
277 case FileAccessMode::WRITE:
278 openMode |= std::ios_base::out | std::ios_base::trunc;
279 break;
280 case FileAccessMode::APPEND:
281 openMode |= std::ios_base::out | std::ios_base::app;
282 break;
283 }
284
285 fh.stream->open(filename, openMode);
286 fh.isOpen = fh.stream->is_open();
287
288 if (fh.isOpen) {
289 filenameToHandle_[filename] = handle;
290 return handle;
291 } else {
292 handles_.erase(handle);
293 throw std::runtime_error("PATH NOT FOUND ERROR");
294 }
295}

549std::vector<uint8_t> FileManager::loadBinaryFile(const std::string &filename,
550 int address) {
551 (void)address; // address is informational only in this host implementation
552 std::ifstream file(filename, std::ios::binary | std::ios::ate);
553 if (!file) {
554 throw std::runtime_error("PATH NOT FOUND ERROR");
555 }
556
557 std::streamsize size = file.tellg();
558 file.seekg(0, std::ios::beg);
559
560 std::vector<uint8_t> buffer(size);
561 if (!file.read(reinterpret_cast<char *>(buffer.data()), size)) {
562 throw std::runtime_error("I/O ERROR");
563 }
564
565 return buffer;
566}
